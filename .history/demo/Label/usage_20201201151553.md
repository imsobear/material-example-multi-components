本 Demo 演示一行文字的用法。

```jsx
import React, { Component } from 'react';
import ReactDOM from 'react-dom';
import { Button } from 'example-component';

class App extends Component {
  render() {
    return (
      <div>
        <Button type="usage" />
      </div>
    );
  }
}

ReactDOM.render((
  <App />
), mountNode);
```
